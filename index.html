Demarrage

1. Importer les fichiers

<!-- Dans votre HTML -->
<link rel="stylesheet" href="tokens/tokens.css">
<link rel="stylesheet" href="components/components.css">

2. Utiliser les classes

<!-- Bouton primaire -->
<button class="nxo-btn nxo-btn-primary">Action</button>

<!-- Card -->
<div class="nxo-card">
  <div class="nxo-card-body">Contenu</div>
</div>

3. Utiliser les tokens CSS

/* Dans votre CSS */
.mon-element {
  color: var(--nxo-text-body);
  background: var(--nxo-card-bg);
  border-radius: var(--nxo-radius-md);
  padding: var(--nxo-spacing-md);
}
