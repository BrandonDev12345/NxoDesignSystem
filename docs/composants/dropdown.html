// ⚠️ Exemple avec Material UI + classes NXO
import React, { useState } from 'react';
import { Button, Menu, MenuItem, Divider } from '@mui/material';
import { Visibility, Edit, ContentCopy, Delete, KeyboardArrowDown } from '@mui/icons-material';

function Dropdown() {
  const [anchorEl, setAnchorEl] = useState(null);
  const open = Boolean(anchorEl);

  const handleClick = (event) => {
    setAnchorEl(event.currentTarget);
  };

  const handleClose = () => {
    setAnchorEl(null);
  };

  return (
    <div className="nxo-dropdown">
      <Button
        variant="outlined"
        className="nxo-btn-secondary"
        onClick={handleClick}
        endIcon={<KeyboardArrowDown />}
      >
        Options
      </Button>
      <Menu
        anchorEl={anchorEl}
        open={open}
        onClose={handleClose}
        className="nxo-dropdown-menu"
      >
        <MenuItem onClick={handleClose} className="nxo-dropdown-item">
          <Visibility fontSize="small" sx={{ mr: 1 }} /> Voir
        </MenuItem>
        <MenuItem onClick={handleClose} className="nxo-dropdown-item">
          <Edit fontSize="small" sx={{ mr: 1 }} /> Modifier
        </MenuItem>
        <MenuItem onClick={handleClose} className="nxo-dropdown-item">
          <ContentCopy fontSize="small" sx={{ mr: 1 }} /> Dupliquer
        </MenuItem>
        <Divider className="nxo-dropdown-divider" />
        <MenuItem onClick={handleClose} className="nxo-dropdown-item nxo-dropdown-item-danger">
          <Delete fontSize="small" sx={{ mr: 1 }} /> Supprimer
        </MenuItem>
      </Menu>
    </div>
  );
}

<script>
document.querySelectorAll('[data-dropdown]').forEach(dropdown => {
  const trigger = dropdown.querySelector('[data-dropdown-trigger]');
  const menu = dropdown.querySelector('[data-dropdown-menu]');

  trigger.addEventListener('click', () => {
    menu.classList.toggle('show');
  });

  document.addEventListener('click', (e) => {
    if (!dropdown.contains(e.target)) {
      menu.classList.remove('show');
    }
  });
});
</script>

import { useState, useRef, useEffect } from 'react';

function UserDropdown({ user }) {
  const [isOpen, setIsOpen] = useState(false);
  const dropdownRef = useRef(null);

  useEffect(() => {
    function handleClickOutside(event) {
      if (dropdownRef.current && !dropdownRef.current.contains(event.target)) {
        setIsOpen(false);
      }
    }
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  return (
    <div className="nxo-dropdown" ref={dropdownRef}>
      <button
        className="nxo-btn nxo-btn-primary"
        onClick={() => setIsOpen(!isOpen)}
      >
        Mon compte <i className="fa-solid fa-chevron-down"></i>
      </button>
      {isOpen && (
        <div className="nxo-dropdown-menu">
          <div className="nxo-dropdown-header">
            <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
              <div className="nxo-avatar nxo-avatar-sm">
                <span className="nxo-avatar-initials">{user.initials}</span>
              </div>
              <div>
                <div style={{ fontWeight: 500 }}>{user.name}</div>
                <div className="nxo-text-sm nxo-text-muted">{user.email}</div>
              </div>
            </div>
          </div>
          <hr className="nxo-dropdown-divider" />
          <a href="#" className="nxo-dropdown-item">
            <i className="fa-solid fa-user"></i> Mon profil
          </a>
          <a href="#" className="nxo-dropdown-item">
            <i className="fa-solid fa-cog"></i> Parametres
          </a>
          <hr className="nxo-dropdown-divider" />
          <a href="#" className="nxo-dropdown-item">
            <i className="fa-solid fa-sign-out-alt"></i> Deconnexion
          </a>
        </div>
      )}
    </div>
  );
}
// user-dropdown.component.ts
import { Component, Input, HostListener, ElementRef } from '@angular/core';

interface User {
  initials: string;
  name: string;
  email: string;
}

@Component({
  selector: 'nxo-user-dropdown',
  templateUrl: './user-dropdown.component.html',
  styleUrls: ['./user-dropdown.component.css']
})
export class UserDropdownComponent {
  @Input() user: User;
  isOpen = false;

  constructor(private elementRef: ElementRef) {}

  toggleDropdown() {
    this.isOpen = !this.isOpen;
  }

  @HostListener('document:click', ['$event'])
  clickOutside(event: Event) {
    if (!this.elementRef.nativeElement.contains(event.target)) {
      this.isOpen = false;
    }
  }
}

Comportement responsive

Le composant Dropdown est naturellement flexible. Le menu deroulant se positionne de maniere absolue et s'adapte au contenu. Aucune adaptation specifique n'est necessaire.
