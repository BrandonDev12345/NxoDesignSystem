// ⚠️ Exemple avec Material UI + classes NXO

import { Chip } from '@mui/material';

function ChipTags() {
  const handleRemove = (tag) => {
    console.log(`Remove ${tag}`);
  };

  return (
    <>
      <Chip
        label="Reseau"
        onDelete={() => handleRemove('Reseau')}
        className="nxo-chip"
      />
      <Chip
        label="Securite"
        onDelete={() => handleRemove('Securite')}
        className="nxo-chip"
      />
      <Chip
        label="Cloud"
        onDelete={() => handleRemove('Cloud')}
        className="nxo-chip nxo-chip-primary"
        color="primary"
      />
    </>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO

import { Component } from '@angular/core';
import { MatChipsModule } from '@angular/material/chips';

@Component({
  selector: 'app-chip-tags',
  template: `
    <mat-chip-listbox>
      <mat-chip class="nxo-chip" (removed)="handleRemove('Reseau')">
        Reseau
        <button matChipRemove><fa-icon icon="times"></fa-icon></button>
      </mat-chip>
      <mat-chip class="nxo-chip" (removed)="handleRemove('Securite')">
        Securite
        <button matChipRemove><fa-icon icon="times"></fa-icon></button>
      </mat-chip>
      <mat-chip class="nxo-chip nxo-chip-primary" (removed)="handleRemove('Cloud')">
        Cloud
        <button matChipRemove><fa-icon icon="times"></fa-icon></button>
      </mat-chip>
    </mat-chip-listbox>
  `
})
export class ChipTagsComponent {
  handleRemove(tag: string) {
    console.log(\`Remove \${tag}\`);
  }
}
{# Twig Template #}
{{ nxo_chip('Reseau', {
  'removable': true,
  'onRemove': 'handleRemoveTag'
}) }}

{{ nxo_chip('Securite', {
  'removable': true,
  'onRemove': 'handleRemoveTag'
}) }}

{{ nxo_chip('Cloud', {
  'variant': 'primary',
  'removable': true,
  'onRemove': 'handleRemoveTag'
}) }}

Comportement responsive

Les Badges sont naturellement flexibles. Ce sont des elements inline qui s'adaptent automatiquement a leur contenu. Aucune adaptation specifique n'est necessaire.

Voir la page Responsive pour tous les details.
