// ⚠️ Exemple avec Material UI + classes NXO
import { Stepper, Step, StepLabel } from '@mui/material';

const steps = ['Informations', 'Verification', 'Paiement', 'Confirmation'];

const HorizontalStepper = () => {
  const activeStep = 1; // 0-indexed, so step 2 is active

  return (
    <Stepper activeStep={activeStep} className="nxo-stepper">
      {steps.map((label) => (
        <Step key={label}>
          <StepLabel className="nxo-stepper-label">{label}</StepLabel>
        </Step>
      ))}
    </Stepper>
  );
};

export default HorizontalStepper;

// ⚠️ Exemple avec Material UI + classes NXO
import { Stepper, Step, StepLabel } from '@mui/material';
import ErrorIcon from '@mui/icons-material/Error';

const steps = ['Etape 1', 'Etape 2', 'Etape 3'];

const StepperWithError = () => {
  const activeStep = 1;
  const errorStep = 1; // 0-indexed

  return (
    <Stepper activeStep={activeStep} className="nxo-stepper">
      {steps.map((label, index) => {
        const isError = index === errorStep;
        return (
          <Step key={label}>
            <StepLabel
              error={isError}
              StepIconComponent={isError ? () => <ErrorIcon color="error" /> : undefined}
              className={`nxo-stepper-label ${isError ? 'error' : ''}`}
            >
              {label}
            </StepLabel>
          </Step>
        );
      })}
    </Stepper>
  );
};

export default StepperWithError;
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatStepperModule, STEPPER_GLOBAL_OPTIONS } from '@angular/material/stepper';
import { MatIconModule } from '@angular/material/icon';

@Component({
  selector: 'app-stepper-error',
  standalone: true,
  imports: [MatStepperModule, MatIconModule],
  providers: [
    { provide: STEPPER_GLOBAL_OPTIONS, useValue: { showError: true } }
  ],
  template: `
    <mat-horizontal-stepper [selectedIndex]="1" class="nxo-stepper">
      @for (step of steps; track step; let i = $index) {
        <mat-step [hasError]="i === errorStep" [label]="step">
          <ng-template matStepperIcon="error">
            <mat-icon>error</mat-icon>
          </ng-template>
        </mat-step>
      }
    </mat-horizontal-stepper>
  `
})
export class StepperErrorComponent {
  steps = ['Etape 1', 'Etape 2', 'Etape 3'];
  errorStep = 1;
}
{# Twig Template #}
{% set steps = ['Etape 1', 'Etape 2', 'Etape 3'] %}
{% set currentStep = 2 %}
{% set errorStep = 2 %}

<div class="nxo-stepper">
  {% for step in steps %}
    {% set stepNumber = loop.index %}
    {% set isCompleted = stepNumber < currentStep and stepNumber != errorStep %}
    {% set isActive = stepNumber == currentStep %}
    {% set hasError = stepNumber == errorStep %}

    <div class="nxo-stepper-item {{ isCompleted ? 'completed' : '' }} {{ isActive ? 'active' : '' }} {{ hasError ? 'error' : '' }}">
      <div class="nxo-stepper-indicator">
        {% if isCompleted %}
          <i class="fa-solid fa-check"></i>
        {% elseif hasError %}
          <i class="fa-solid fa-exclamation"></i>
        {% else %}
          {{ stepNumber }}
        {% endif %}
      </div>
      <div class="nxo-stepper-label">{{ step }}</div>
    </div>
  {% endfor %}
</div>

Comportement responsive

Le stepper passe automatiquement en disposition verticale sur mobile pour une meilleure lisibilite.
Breakpoint Comportement
≤ 768px .nxo-stepper passe en flex-direction: column. Chaque etape s'affiche en ligne (icone + texte) au lieu d'en colonne
