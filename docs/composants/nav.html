// ⚠️ Exemple avec Material UI + classes NXO
import { Box, Link } from '@mui/material';
import { useState } from 'react';

const Navigation = () => {
  const [activeLink, setActiveLink] = useState('accueil');

  const navItems = ['accueil', 'produits', 'services', 'contact'];

  return (
    <Box component="nav" className="nxo-nav" sx={{ display: 'flex', gap: 2 }}>
      {navItems.map(item => (
        <Link
          key={item}
          href="#"
          className={`nxo-nav-link ${activeLink === item ? 'active' : ''}`}
          onClick={() => setActiveLink(item)}
          underline="none"
        >
          {item.charAt(0).toUpperCase() + item.slice(1)}
        </Link>
      ))}
    </Box>
  );
};
// ⚠️ Exemple avec Angular Material + classes NXO

        {{ item.label }}

export class NavigationComponent {

}

// ⚠️ Exemple avec Material UI + classes NXO
import { Box, Link } from '@mui/material';
import { Home, Inventory, Settings, Email } from '@mui/icons-material';
import { useState } from 'react';

const IconNavigation = () => {
  const [activeLink, setActiveLink] = useState('accueil');

  const navItems = [
    { id: 'accueil', label: 'Accueil', icon: <Home fontSize="small" /> },
    { id: 'produits', label: 'Produits', icon: <Inventory fontSize="small" /> },
    { id: 'services', label: 'Services', icon: <Settings fontSize="small" /> },
    { id: 'contact', label: 'Contact', icon: <Email fontSize="small" /> }
  ];

  return (
    <Box component="nav" className="nxo-nav" sx={{ display: 'flex', gap: 2 }}>
      {navItems.map(item => (
        <Link
          key={item.id}
          href="#"
          className={`nxo-nav-link ${activeLink === item.id ? 'active' : ''}`}
          onClick={() => setActiveLink(item.id)}
          sx={{ display: 'flex', alignItems: 'center', gap: 0.5 }}
          underline="none"
        >
          {item.icon} {item.label}
        </Link>
      ))}
    </Box>
  );
};

// ⚠️ Exemple avec Material UI + classes NXO
import { List, ListItemButton, ListItemText } from '@mui/material';
import { useState } from 'react';

const VerticalNavigation = () => {
  const [activeLink, setActiveLink] = useState('dashboard');

  const menuItems = [
    { id: 'dashboard', label: 'Dashboard' },
    { id: 'utilisateurs', label: 'Utilisateurs' },
    { id: 'rapports', label: 'Rapports' },
    { id: 'parametres', label: 'Parametres' }
  ];

  return (
    <List component="nav" className="nxo-nav" sx={{ flexDirection: 'column', width: 200 }}>
      {menuItems.map(item => (
        <ListItemButton
          key={item.id}
          selected={activeLink === item.id}
          onClick={() => setActiveLink(item.id)}
          className={`nxo-nav-link ${activeLink === item.id ? 'active' : ''}`}
        >
          <ListItemText primary={item.label} />
        </ListItemButton>
      ))}
    </List>
  );
};
// ⚠️ Exemple avec Angular Material + classes NXO

≤ 768px .nxo-nav active overflow-x: auto avec flex-wrap: nowrap et scrollbar masquee
