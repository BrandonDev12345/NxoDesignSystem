// ⚠️ Exemple avec Material UI + classes NXO
import { Tabs, Tab, Box, Typography } from '@mui/material';
import { useState } from 'react';

function BasicTabs() {
  const [value, setValue] = useState(0);

  const tabs = [
    { label: 'Onglet 1', content: 'Contenu de l\'onglet actif.' },
    { label: 'Onglet 2', content: 'Contenu de l\'onglet 2.' },
    { label: 'Onglet 3', content: 'Contenu de l\'onglet 3.' }
  ];

  return (
    <Box className="nxo-tabs-container">
      <Tabs value={value} onChange={(e, newValue) => setValue(newValue)} className="nxo-tabs">
        {tabs.map((tab, index) => (
          <Tab key={index} label={tab.label} className="nxo-tab" />
        ))}
        <Tab label="Desactive" disabled className="nxo-tab" />
      </Tabs>
      <Box className="nxo-tab-content" sx={{ p: 2, border: 1, borderColor: 'divider', borderTop: 0 }}>
        <Typography className="nxo-text">{tabs[value].content}</Typography>
      </Box>
    </Box>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatTabsModule } from '@angular/material/tabs';

@Component({
  selector: 'app-basic-tabs',
  template: `
    <mat-tab-group class="nxo-tabs" (selectedTabChange)="onTabChange($event)">
      <mat-tab *ngFor="let tab of tabs" [label]="tab.label" class="nxo-tab">
        <div class="nxo-tab-content">
          <p class="nxo-text">{{ tab.content }}</p>
        </div>
      </mat-tab>
      <mat-tab label="Desactive" disabled class="nxo-tab"></mat-tab>
    </mat-tab-group>
  `
})
export class BasicTabsComponent {
  tabs = [
    { label: 'Onglet 1', content: 'Contenu de l\'onglet actif.' },
    { label: 'Onglet 2', content: 'Contenu de l\'onglet 2.' },
    { label: 'Onglet 3', content: 'Contenu de l\'onglet 3.' }
  ];

  onTabChange(event: any) {
    console.log('Tab changed:', event.index);
  }
}

// ⚠️ Exemple avec Material UI + classes NXO
import { Tabs, Tab, Badge, Chip } from '@mui/material';
import { useState } from 'react';

function BadgeTabs() {
  const [value, setValue] = useState(0);

  const tabs = [
    { label: 'Tous', count: 42, variant: 'default' },
    { label: 'En attente', count: 5, variant: 'warning' },
    { label: 'Valides', count: 37, variant: 'success' }
  ];

  return (
    <Tabs value={value} onChange={(e, v) => setValue(v)} className="nxo-tabs">
      {tabs.map((tab, index) => (
        <Tab
          key={index}
          className="nxo-tab"
          label={
            <>
              {tab.label}
              <Chip size="small" label={tab.count} className={`nxo-badge nxo-badge-${tab.variant}`} sx={{ ml: 1 }} />
            </>
          }
        />
      ))}
    </Tabs>
  );
}

Comportement responsive

Les onglets deviennent scrollables horizontalement sur mobile. La barre de scroll est masquee pour un rendu propre.
Breakpoint Comportement
≤ 768px .nxo-tabs active overflow-x: auto avec scrollbar masquee. Chaque .nxo-tab garde sa largeur naturelle (flex-shrink: 0)
