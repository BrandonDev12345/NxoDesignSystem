import { Dialog, DialogTitle, DialogContent, DialogActions, Button, IconButton } from '@mui/material';
import { Close as CloseIcon } from '@mui/icons-material';
import { useState } from 'react';

function Example() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <>
      <Button variant="contained" className="nxo-btn nxo-btn-primary" onClick={() => setIsOpen(true)}>
        Ouvrir la modale
      </Button>

      <Dialog open={isOpen} onClose={() => setIsOpen(false)} className="nxo-modal">
        <DialogTitle className="nxo-modal-header">
          Titre de la modale
          <IconButton onClick={() => setIsOpen(false)} className="nxo-modal-close">
            <CloseIcon />
          </IconButton>
        </DialogTitle>
        <DialogContent className="nxo-modal-body">Contenu de la modale...</DialogContent>
        <DialogActions className="nxo-modal-footer">
          <Button className="nxo-btn nxo-btn-ghost" onClick={() => setIsOpen(false)}>
            Annuler
          </Button>
          <Button variant="contained" className="nxo-btn nxo-btn-primary">Confirmer</Button>
        </DialogActions>
      </Dialog>
    </>
  );
}

// ⚠️ Exemple avec Material UI + classes NXO
import { Dialog, DialogContent, DialogActions, Button, Box, Typography } from '@mui/material';
import { Warning as WarningIcon } from '@mui/icons-material';
import { useState } from 'react';

function Example() {
  const [isOpen, setIsOpen] = useState(false);

  const handleConfirm = () => {
    // Logique de suppression
    setIsOpen(false);
  };

  return (
    <>
      <Button variant="contained" color="error" className="nxo-btn nxo-btn-danger" onClick={() => setIsOpen(true)}>
        Supprimer
      </Button>

      <Dialog open={isOpen} onClose={() => setIsOpen(false)} className="nxo-modal" maxWidth="xs">
        <DialogContent className="nxo-modal-body" sx={{ textAlign: 'center', py: 4 }}>
          <Box className="nxo-modal-icon nxo-modal-icon--danger" sx={{ mb: 2 }}>
            <WarningIcon fontSize="large" />
          </Box>
          <Typography variant="h6">Confirmer la suppression ?</Typography>
          <Typography color="text.secondary">Cette action est irreversible.</Typography>
        </DialogContent>
        <DialogActions className="nxo-modal-footer" sx={{ justifyContent: 'center' }}>
          <Button className="nxo-btn nxo-btn-ghost" onClick={() => setIsOpen(false)}>Annuler</Button>
          <Button variant="contained" color="error" className="nxo-btn nxo-btn-danger" onClick={handleConfirm}>
            Supprimer
          </Button>
        </DialogActions>
      </Dialog>
    </>
  );
}

// ⚠️ Exemple avec Material UI + classes NXO
import { Dialog, DialogTitle, DialogContent, DialogActions, Button, TextField, FormControl, InputLabel, Select, MenuItem, IconButton } from '@mui/material';
import { Close as CloseIcon } from '@mui/icons-material';
import { useState } from 'react';

function UserFormModal({ isOpen, onClose, onSubmit }) {
  const [formData, setFormData] = useState({ name: '', email: '', role: '' });

  return (
    <Dialog open={isOpen} onClose={onClose} className="nxo-modal">
      <DialogTitle className="nxo-modal-header">
        Nouvel utilisateur
        <IconButton onClick={onClose} className="nxo-modal-close"><CloseIcon /></IconButton>
      </DialogTitle>
      <DialogContent className="nxo-modal-body">
        <div className="nxo-form-group">
          <TextField
            label="Nom complet"
            className="nxo-input"
            fullWidth
            value={formData.name}
            onChange={e => setFormData({...formData, name: e.target.value})}
            placeholder="Jean Dupont"
          />
        </div>
        <div className="nxo-form-group">
          <TextField label="Email" type="email" className="nxo-input" fullWidth placeholder="jean@nxo.fr" />
        </div>
        <div className="nxo-form-group">
          <FormControl fullWidth className="nxo-select">
            <InputLabel>Role</InputLabel>
            <Select value={formData.role} onChange={e => setFormData({...formData, role: e.target.value})}>
              <MenuItem value="admin">Administrateur</MenuItem>
              <MenuItem value="user">Utilisateur</MenuItem>
            </Select>
          </FormControl>
        </div>
      </DialogContent>
      <DialogActions className="nxo-modal-footer">
        <Button className="nxo-btn nxo-btn-ghost" onClick={onClose}>Annuler</Button>
        <Button variant="contained" className="nxo-btn nxo-btn-primary" onClick={() => onSubmit(formData)}>Creer</Button>
      </DialogActions>
    </Dialog>
  );
}

JavaScript

// Ouvrir une modale
document.getElementById('myModal').classList.add('open');

// Fermer une modale
document.getElementById('myModal').classList.remove('open');

// Fermer au clic sur backdrop
document.querySelectorAll('.nxo-modal-backdrop').forEach(backdrop => {
  backdrop.addEventListener('click', () => {
    backdrop.parentElement.classList.remove('open');
  });
});

// Fermer avec la touche Escape
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') {
    document.querySelectorAll('.nxo-modal.open').forEach(modal => {
      modal.classList.remove('open');
    });
  }
});

Comportement responsive

Les modales passent en plein ecran sur mobile pour une meilleure lisibilite. Les boutons du footer sont empiles verticalement.
Breakpoint Comportement
≤ 768px La modale occupe 100% de l'ecran (width: 100%, height: 100vh, border-radius: 0). Les boutons du .nxo-modal-footer passent en colonne
