// ⚠️ Exemple avec Angular Material + classes NXO
// Note: Angular Material n'a pas de composant Avatar natif
// Utilisez les classes NXO avec un composant personnalise
import { Component, Input } from '@angular/core';

@Component({
  selector: 'nxo-avatar',
  template: `
    <div [class]="getAvatarClass()" [ngStyle]="customStyle">
      <span class="nxo-avatar-initials">{{ initials }}</span>
    </div>
  `
})
export class AvatarComponent {
  @Input() size: 'xs' | 'sm' | 'md' | 'lg' | 'xl' = 'md';
  @Input() initials: string = '';
  @Input() customStyle: any = {};

  getAvatarClass(): string {
    return this.size !== 'md' ? `nxo-avatar nxo-avatar-${this.size}` : 'nxo-avatar';
  }
}

import { Avatar, AvatarGroup } from '@mui/material';

const AvatarGroupExample = () => {
  const users = [
    { initials: 'AB', bgcolor: 'var(--fayat-blue)' },
    { initials: 'CD', bgcolor: 'var(--nxo-color-success)' },
    { initials: 'EF', bgcolor: 'var(--nxo-color-warning)' },
    { initials: 'GH', bgcolor: 'var(--nxo-color-error)' },
    { initials: 'IJ', bgcolor: 'var(--nxo-color-info)' },
    { initials: 'KL', bgcolor: 'var(--fayat-blue)' }
  ];

  return (
    <AvatarGroup max={4} className="nxo-avatar-group">
      {users.map((user, index) => (
        <Avatar key={index} className="nxo-avatar nxo-avatar-sm" sx={{ bgcolor: user.bgcolor, width: 32, height: 32 }}>
          {user.initials}
        </Avatar>
      ))}
    </AvatarGroup>
  );
};
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component, Input } from '@angular/core';
// Note: Angular Material n'a pas d'AvatarGroup, utilisez les classes NXO

interface AvatarData { initials: string; backgroundColor: string; }

@Component({
  selector: 'nxo-avatar-group',
  template: `<div class="nxo-avatar-group">
    <div *ngFor="let avatar of displayAvatars" [class]="getAvatarClass()" [style.background-color]="avatar.backgroundColor">
      <span class="nxo-avatar-initials">{{ avatar.initials }}</span>
    </div>
    <div *ngIf="remaining > 0" [class]="getAvatarClass()" style="background-color: var(--nxo-text-muted)">
      <span class="nxo-avatar-initials">+{{ remaining }}</span>
    </div>
  </div>`
})
export class AvatarGroupComponent {
  @Input() avatars: AvatarData[] = [];
  @Input() maxDisplay: number = 3;
  @Input() size: 'xs' | 'sm' | 'md' | 'lg' | 'xl' = 'sm';

  get displayAvatars() { return this.avatars.slice(0, this.maxDisplay); }
  get remaining() { return Math.max(0, this.avatars.length - this.maxDisplay); }
  getAvatarClass() { return this.size !== 'md' ? `nxo-avatar nxo-avatar-${this.size}` : 'nxo-avatar'; }
}

// Usage
<nxo-avatar-group [avatars]="users" [maxDisplay]="3" size="sm"></nxo-avatar-group>
{# avatar-group.html.twig #}
{% macro avatar_group(avatars, maxDisplay = 3, size = 'sm') %}
  <div class="nxo-avatar-group">
    {% for avatar in avatars|slice(0, maxDisplay) %}
      <div class="nxo-avatar{% if size != 'md' %} nxo-avatar-{{ size }}{% endif %}"
           style="background-color: {{ avatar.backgroundColor }};">
        <span class="nxo-avatar-initials">{{ avatar.initials }}</span>
      </div>
    {% endfor %}

    {% set remaining = avatars|length - maxDisplay %}
    {% if remaining > 0 %}
      <div class="nxo-avatar{% if size != 'md' %} nxo-avatar-{{ size }}{% endif %}"
           style="background-color: var(--nxo-text-muted);">
        <span class="nxo-avatar-initials">+{{ remaining }}</span>
      </div>
    {% endif %}
  </div>
{% endmacro %}

{# Usage #}
{% import 'avatar-group.html.twig' as avatar_macro %}

{% set users = [
  { initials: 'AB', backgroundColor: 'var(--fayat-blue)' },
  { initials: 'CD', backgroundColor: 'var(--nxo-color-success)' },
  { initials: 'EF', backgroundColor: 'var(--nxo-color-warning)' }
] %}

{{ avatar_macro.avatar_group(users, 3, 'sm') }}

Comportement responsive

Les Avatars sont naturellement flexibles. Leurs dimensions fixes (cercles) s'integrent dans tous les contextes sans adaptation. Aucune adaptation specifique n'est necessaire.

Voir la page Responsive pour tous les details.
