import React from 'react';
import { Drawer, List, ListItem, ListItemButton, ListItemIcon, ListItemText, Box } from '@mui/material';
import { Home, People, BarChart, Settings } from '@mui/icons-material';

const Sidebar = () => {
  const [activeLink, setActiveLink] = React.useState('dashboard');

  const navItems = [
    { id: 'dashboard', icon: Home, label: 'Dashboard' },
    { id: 'users', icon: People, label: 'Utilisateurs' },
    { id: 'reports', icon: BarChart, label: 'Rapports' },
    { id: 'settings', icon: Settings, label: 'Parametres' }
  ];

  return (
    <Drawer
      variant="permanent"
      className="nxo-sidebar"
      sx={{ width: 250 }}
    >
      <Box className="nxo-sidebar-header" sx={{ p: 2 }}>
        <span>Mon Application</span>
      </Box>
      <List className="nxo-sidebar-nav">
        {navItems.map(item => (
          <ListItem key={item.id} disablePadding>
            <ListItemButton
              className={`nxo-sidebar-link ${activeLink === item.id ? 'active' : ''}`}
              onClick={() => setActiveLink(item.id)}
            >
              <ListItemIcon>
                <item.icon />
              </ListItemIcon>
              <ListItemText primary={item.label} />
            </ListItemButton>
          </ListItem>
        ))}
      </List>
    </Drawer>
  );
};

export default Sidebar;

const SidebarWithSections = () => {
  const [activeLink, setActiveLink] = React.useState('dashboard');
  const [messageCount] = React.useState(3);

  const sections = [
    {
      title: 'Principal',
      items: [
        { id: 'dashboard', icon: Home, label: 'Dashboard' },
        { id: 'messages', icon: Inbox, label: 'Messages', badge: messageCount }
      ]
    },
    {
      title: 'Gestion',
      items: [
        { id: 'teams', icon: Users, label: 'Equipes' },
        { id: 'projects', icon: Folder, label: 'Projets' }
      ]
    }
  ];

  return (
    <aside className="nxo-sidebar">
      <div className="nxo-sidebar-header">
        <span>NXO Portal</span>
      </div>
      <nav className="nxo-sidebar-nav">
        {sections.map((section, idx) => (
          <div key={idx} className="nxo-sidebar-section">
            <div className="nxo-sidebar-section-title">
              {section.title}
            </div>
            {section.items.map(item => (
              <a
                key={item.id}
                href="#"
                className={`nxo-sidebar-link ${activeLink === item.id ? 'active' : ''}`}
                onClick={(e) => {
                  e.preventDefault();
                  setActiveLink(item.id);
                }}
              >
                <item.icon />
                <span>{item.label}</span>
                {item.badge && (
                  <span className="nxo-badge nxo-badge-error">
                    {item.badge}
                  </span>
                )}
              </a>
            ))}
          </div>
        ))}
      </nav>
    </aside>
  );
};

<script>
  const toggle = document.getElementById('sidebarToggle');
  const sidebar = document.getElementById('sidebar');
  const overlay = document.getElementById('sidebarOverlay');

  // Ouvrir / fermer la sidebar
  toggle.addEventListener('click', () => {
    sidebar.classList.toggle('open');
    overlay.classList.toggle('open');
  });

  // Fermer en cliquant sur l'overlay
  overlay.addEventListener('click', () => {
    sidebar.classList.remove('open');
    overlay.classList.remove('open');
  });
</script>

const ResponsiveSidebar = () => {

Classes CSS

Classe Description
.nxo-sidebar Conteneur principal de la sidebar
.nxo-sidebar-header En-tete de la sidebar (logo, nom de l'app)
.nxo-sidebar-nav Conteneur de la navigation
.nxo-sidebar-link Lien de navigation. Ajouter .active pour l'etat actif.
.nxo-sidebar-section Groupe de liens avec un titre
.nxo-sidebar-section-title Titre d'une section
.nxo-sidebar-toggle Bouton hamburger (visible en mobile < 768px)
.nxo-sidebar-overlay Overlay sombre derriere la sidebar en mobile
.open Classe a ajouter via JS sur .nxo-sidebar, .nxo-sidebar-overlay pour ouvrir

Comportement responsive

Breakpoint : 768px
En dessous de 768px, la sidebar passe en position: fixed et se cache via transform: translateX(-100%). Le bouton .nxo-sidebar-toggle devient visible automatiquement. Il suffit d'ajouter la classe .open via JavaScript pour l'afficher.
