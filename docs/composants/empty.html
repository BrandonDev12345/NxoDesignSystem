Empty State

Etats vides pour les listes, tableaux ou sections sans contenu.

Preview

Avec action

Preview

Aucun projet

Vous n'avez pas encore cree de projet. Commencez par en creer un nouveau.

<div class="nxo-empty-state">
  <div class="nxo-empty-state-icon" style="color: var(--fayat-blue);">
    <i class="fa-solid fa-folder-open"></i>
  </div>
  <h3 class="nxo-empty-state-title">Aucun projet</h3>
  <p class="nxo-empty-state-description">Vous n'avez pas encore cree de projet. Commencez par en creer un nouveau.</p>
  <button class="nxo-btn nxo-btn-primary">
    <i class="fa-solid fa-plus"></i> Nouveau projet
  </button>
</div>
// ⚠️ Exemple avec Material UI + classes NXO
import { Box, Typography, Button } from '@mui/material';

// Empty state avec action
<Box className="nxo-empty-state">
  <Box className="nxo-empty-state-icon" sx={{ color: 'var(--fayat-blue)' }}>
    <i className="fa-solid fa-folder-open"></i>
  </Box>
  <Typography variant="h6" component="h3" className="nxo-empty-state-title">
    Aucun projet
  </Typography>
  <Typography variant="body2" className="nxo-empty-state-description">
    Vous n'avez pas encore cree de projet. Commencez par en creer un nouveau.
  </Typography>
  <Button variant="contained" className="nxo-btn nxo-btn-primary" onClick={handleNewProject}>
    <i className="fa-solid fa-plus"></i> Nouveau projet
  </Button>
</Box>
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatButtonModule } from '@angular/material/button';

@Component({
  selector: 'app-empty-state-with-action',
  standalone: true,
  imports: [MatButtonModule],
  template: `
    <div class="nxo-empty-state">
      <div class="nxo-empty-state-icon" style="color: var(--fayat-blue);">
        <i class="fa-solid fa-folder-open"></i>
      </div>
      <h3 class="nxo-empty-state-title">Aucun projet</h3>
      <p class="nxo-empty-state-description">
        Vous n'avez pas encore cree de projet. Commencez par en creer un nouveau.
      </p>
      <button mat-raised-button class="nxo-btn nxo-btn-primary" (click)="handleNewProject()">
        <i class="fa-solid fa-plus"></i> Nouveau projet
      </button>
    </div>
  `
})
export class EmptyStateWithActionComponent {
  handleNewProject() {
    console.log('New project');
  }
}

  </button>
