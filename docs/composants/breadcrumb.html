Breadcrumb

Fil d'Ariane pour indiquer la position de l'utilisateur dans la hierarchie du site.

Breadcrumb basique

Preview
Accueil Produits Detail produit 
<nav class="nxo-breadcrumb">
  <span class="nxo-breadcrumb-item"><a href="#">Accueil</a></span>
  <span class="nxo-breadcrumb-item"><a href="#">Produits</a></span>
  <span class="nxo-breadcrumb-item active">Detail produit</span>
</nav>
// ⚠️ Exemple avec Material UI + classes NXO
import { Breadcrumbs, Link, Typography } from '@mui/material';
import { NavigateNext as NavigateNextIcon } from '@mui/icons-material';

const BreadcrumbExample = () => {
  const items = [
    { label: 'Accueil', href: '#' },
    { label: 'Produits', href: '#' },
    { label: 'Detail produit', active: true }
  ];

  return (
    <Breadcrumbs
      separator={<NavigateNextIcon fontSize="small" />}
      className="nxo-breadcrumb"
    >
      {items.map((item, index) => (
        item.active ? (
          <Typography key={index} className="nxo-breadcrumb-item active">
            {item.label}
          </Typography>
        ) : (
          <Link key={index} href={item.href} className="nxo-breadcrumb-item">
            {item.label}
          </Link>
        )
      ))}
    </Breadcrumbs>
  );
};

export default BreadcrumbExample;
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component, Input } from '@angular/core';
// Note: Angular Material n'a pas de composant Breadcrumb natif
// Utilisez les classes NXO avec un composant personnalise

interface BreadcrumbItem {
  label: string;
  href?: string;
  active?: boolean;
}

@Component({
  selector: 'app-breadcrumb',
  template: `
    <nav class="nxo-breadcrumb">
      <span
        *ngFor="let item of items"
        [class.active]="item.active"
        class="nxo-breadcrumb-item"
      >
        <a *ngIf="!item.active" [routerLink]="item.href">{{ item.label }}</a>
        <ng-container *ngIf="item.active">{{ item.label }}</ng-container>
      </span>
    </nav>
  `
})
export class BreadcrumbComponent {
  @Input() items: BreadcrumbItem[] = [
    { label: 'Accueil', href: '/' },
    { label: 'Produits', href: '/produits' },
    { label: 'Detail produit', active: true }
  ];
}
{# breadcrumb.twig #}
{% set breadcrumb_items = [
  { label: 'Accueil', href: '#' },
  { label: 'Produits', href: '#' },
  { label: 'Detail produit', active: true }
] %}

<nav class="nxo-breadcrumb">
  {% for item in breadcrumb_items %}
    <span class="nxo-breadcrumb-item {{ item.active ? 'active' : '' }}">
      {% if item.active %}
        {{ item.label }}
      {% else %}
        <a href="{{ item.href }}">{{ item.label }}</a>
      {% endif %}
    </span>
  {% endfor %}
</nav>

Comportement responsive

Le fil d'Ariane devient scrollable horizontalement sur mobile. La barre de scroll est masquee pour un rendu propre.
Breakpoint Comportement
≤ 768px .nxo-breadcrumb active overflow-x: auto avec flex-wrap: nowrap et scrollbar masquee
