import React, { useRef, useState } from 'react';
import { Button } from '@mui/material';
import CloudUploadIcon from '@mui/icons-material/CloudUpload';

function Dropzone() {
  const [isDragging, setIsDragging] = useState(false);
  const fileInputRef = useRef(null);

  const handleDragOver = (e) => {
    e.preventDefault();
    setIsDragging(true);
  };

  const handleDragLeave = () => {
    setIsDragging(false);
  };

  const handleDrop = (e) => {
    e.preventDefault();
    setIsDragging(false);
    const files = e.dataTransfer.files;
    // Handle files
    console.log(files);
  };

  const handleClick = () => {
    fileInputRef.current?.click();
  };

  const handleFileChange = (e) => {
    const files = e.target.files;
    // Handle files
    console.log(files);
  };

  return (
    <div
      className={`nxo-dropzone ${isDragging ? 'dragging' : ''}`}
      onDragOver={handleDragOver}
      onDragLeave={handleDragLeave}
      onDrop={handleDrop}
      onClick={handleClick}
    >
      <div className="nxo-dropzone-icon">
        <CloudUploadIcon sx={{ fontSize: 48 }} />
      </div>
      <div className="nxo-dropzone-text">
        <p><strong>Glissez vos fichiers ici</strong></p>
        <p className="nxo-text-sm nxo-text-muted">ou cliquez pour parcourir</p>
      </div>
      <input
        type="file"
        className="nxo-dropzone-input"
        multiple
        ref={fileInputRef}
        onChange={handleFileChange}
      />
    </div>
  );
}

// ⚠️ Exemple avec Material UI + classes NXO
import React, { useRef, useState } from 'react';
import { Button } from '@mui/material';
import AttachFileIcon from '@mui/icons-material/AttachFile';

function FileInput() {
  const [fileName, setFileName] = useState('Aucun fichier selectionne');
  const fileInputRef = useRef(null);

  return (
    <div className="nxo-file-input">
      <input type="file" ref={fileInputRef} style={{ display: 'none' }}
             onChange={(e) => setFileName(e.target.files?.[0]?.name || 'Aucun fichier')} />
      <Button variant="outlined" className="nxo-btn-secondary"
              startIcon={<AttachFileIcon />} onClick={() => fileInputRef.current?.click()}>
        Choisir un fichier
      </Button>
      <span className="nxo-text-sm nxo-text-muted" style={{ marginLeft: '12px' }}>{fileName}</span>
    </div>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component, ElementRef, ViewChild } from '@angular/core';

@Component({
  selector: 'app-file-input',
  template: `
    <div class="nxo-file-input">
      <input #fileInput type="file" style="display: none" (change)="onFileChange($event)">
      <button class="nxo-btn nxo-btn-secondary" (click)="fileInput.click()">
        <i class="fa-solid fa-paperclip"></i> Choisir un fichier
      </button>
      <span class="nxo-text-sm nxo-text-muted" style="margin-left: 12px">{{ fileName }}</span>
    </div>
  `
})
export class FileInputComponent {
  fileName = 'Aucun fichier selectionne';
  onFileChange(event: Event) {
    const file = (event.target as HTMLInputElement).files?.[0];
    if (file) this.fileName = file.name;
  }
}

    <span class="nxo-text-sm nxo-text-muted" style="margin-left: 12px;">Aucun fichier selectionne</span>

import React, { useState } from 'react';

interface FileItem {
  id: string;
  name: string;
  size: string;
  icon: string;
  color: string;
}

function FileList() {
  const [files, setFiles] = useState<FileItem[]>([
    { id: '1', name: 'document.pdf', size: '2.4 MB', icon: 'fa-file-pdf', color: 'var(--nxo-color-error)' },
    { id: '2', name: 'rapport.xlsx', size: '856 KB', icon: 'fa-file-excel', color: 'var(--nxo-color-success)' },
    { id: '3', name: 'photo.jpg', size: '1.2 MB', icon: 'fa-file-image', color: 'var(--fayat-blue)' }
  ]);

  const handleRemove = (id: string) => {
    setFiles(files.filter(file => file.id !== id));
  };

  return (
    <div className="nxo-file-list">
      {files.map(file => (
        <div key={file.id} className="nxo-file-item">
          <div className="nxo-file-item-icon" style={{ color: file.color }}>
            <i className={`fa-solid ${file.icon}`}></i>
          </div>
          <div className="nxo-file-item-info">
            <div className="nxo-file-item-name">{file.name}</div>
            <div className="nxo-file-item-size">{file.size}</div>
          </div>
          <button
            className="nxo-file-item-remove"
            onClick={() => handleRemove(file.id)}
          >
            <i className="fa-solid fa-times"></i>
          </button>
        </div>
      ))}
    </div>
  );
}
import { Component } from '@angular/core';

interface FileItem {
  id: string;
  name: string;
  size: string;
  icon: string;
  color: string;
}

@Component({
  selector: 'app-file-list',
  template: `
    <div class="nxo-file-list">
      <div *ngFor="let file of files" class="nxo-file-item">
        <div class="nxo-file-item-icon" [style.color]="file.color">
          <i [class]="'fa-solid ' + file.icon"></i>
        </div>
        <div class="nxo-file-item-info">
          <div class="nxo-file-item-name">{{ file.name }}</div>
          <div class="nxo-file-item-size">{{ file.size }}</div>
        </div>
        <button class="nxo-file-item-remove" (click)="removeFile(file.id)">
          <i class="fa-solid fa-times"></i>
        </button>
      </div>
    </div>
  `
})
export class FileListComponent {
  files: FileItem[] = [
    { id: '1', name: 'document.pdf', size: '2.4 MB', icon: 'fa-file-pdf', color: 'var(--nxo-color-error)' },
    { id: '2', name: 'rapport.xlsx', size: '856 KB', icon: 'fa-file-excel', color: 'var(--nxo-color-success)' },
    { id: '3', name: 'photo.jpg', size: '1.2 MB', icon: 'fa-file-image', color: 'var(--fayat-blue)' }
  ];

  removeFile(id: string): void {
    this.files = this.files.filter(file => file.id !== id);
  }
}
