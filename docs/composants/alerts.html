// ⚠️ Exemple avec Material UI + classes NXO

import { Alert, IconButton, Collapse } from '@mui/material';
import { useState } from 'react';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import { faTimes } from '@fortawesome/free-solid-svg-icons';

function DismissibleAlert() {
  const [isVisible, setIsVisible] = useState(true);

  return (
    <Collapse in={isVisible}>
      <Alert
        severity="info"
        className="nxo-alert"
        action={
          <IconButton size="small" onClick={() => setIsVisible(false)}>
            <FontAwesomeIcon icon={faTimes} />
          </IconButton>
        }
      >
        Cette alerte peut etre fermee par l'utilisateur.
      </Alert>
    </Collapse>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO

import { Component } from '@angular/core';
import { FontAwesomeModule } from '@fortawesome/angular-fontawesome';

@Component({
  selector: 'app-dismissible-alert',
  template: `
    <div class="nxo-alert nxo-alert-info" *ngIf="isVisible">
      <fa-icon icon="info-circle" class="nxo-alert-icon"></fa-icon>
      <div class="nxo-alert-content">
        Cette alerte peut etre fermee par l'utilisateur.
      </div>
      <button class="nxo-alert-close" (click)="handleClose()">
        <fa-icon icon="times"></fa-icon>
      </button>
    </div>
  `
})
export class DismissibleAlertComponent {
  isVisible = true;

  handleClose() {
    this.isVisible = false;
  }
}

import { Alert, Button } from '@mui/material';

function AlertWithAction() {
  const handleExtend = () => {
    console.log('Session prolongee');
  };

  return (
    <Alert
      severity="warning"
      className="nxo-alert nxo-alert-warning"
      action={
        <Button
          color="warning"
          size="small"
          className="nxo-btn nxo-btn-warning nxo-btn-sm"
          onClick={handleExtend}
        >
          Prolonger
        </Button>
      }

      <strong>Session expirante</strong><br />

    </Alert>
  );
}

// ⚠️ Exemple avec Material UI + classes NXO

import { Snackbar, Alert } from '@mui/material';
import { useState } from 'react';

function ToastExample() {
  const [open, setOpen] = useState(false);
  const [message, setMessage] = useState('');
  const [severity, setSeverity] = useState('success');

  const showToast = (msg, sev) => {
    setMessage(msg);
    setSeverity(sev);
    setOpen(true);
  };

  return (
    <>
      <button onClick={() => showToast('Fichier enregistre avec succes', 'success')}>
        Show Success
      </button>
      <button onClick={() => showToast('Erreur lors de l\'envoi', 'error')}>
        Show Error
      </button>
      <button onClick={() => showToast('Nouvelle mise a jour disponible', 'info')}>
        Show Info
      </button>

      <Snackbar open={open} autoHideDuration={4000} onClose={() => setOpen(false)}>
        <Alert severity={severity} className="nxo-toast">{message}</Alert>
      </Snackbar>
    </>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO

import { Component } from '@angular/core';
import { MatSnackBar, MatSnackBarModule } from '@angular/material/snack-bar';

@Component({
  selector: 'app-toast-example',
  template: `
    <button (click)="showSuccess()">Show Success</button>
    <button (click)="showError()">Show Error</button>
    <button (click)="showInfo()">Show Info</button>
  `
})
export class ToastExampleComponent {
  constructor(private snackBar: MatSnackBar) {}

  showSuccess() {
    this.snackBar.open('Fichier enregistre avec succes', 'Fermer', {
      duration: 4000,
      panelClass: ['nxo-toast', 'nxo-toast-success']
    });
  }

  showError() {
    this.snackBar.open('Erreur lors de l\'envoi', 'Fermer', {
      duration: 4000,
      panelClass: ['nxo-toast', 'nxo-toast-error']
    });
  }

  showInfo() {
    this.snackBar.open('Nouvelle mise a jour disponible', 'Fermer', {
      duration: 4000,
      panelClass: ['nxo-toast', 'nxo-toast-info']
    });
  }
}
