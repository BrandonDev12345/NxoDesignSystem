import { Pagination as MuiPagination } from '@mui/material';
import { useState } from 'react';

function PaginationExample() {
  const [page, setPage] = useState(1);
  const totalPages = 10;

  return (
    <MuiPagination
      count={totalPages}
      page={page}
      onChange={(event, value) => setPage(value)}
      className="nxo-pagination"
      color="primary"
      showFirstButton
      showLastButton
    />
  );
}

// Alternative avec classes NXO personnalisees
function CustomPagination() {
  const [currentPage, setCurrentPage] = useState(1);

  return (
    <nav className="nxo-pagination">
      <span className={`nxo-pagination-item ${currentPage === 1 ? 'disabled' : ''}`}>
        <a className="nxo-pagination-link" onClick={() => setCurrentPage(p => Math.max(1, p - 1))}>
          <i className="fa-solid fa-chevron-left"></i>
        </a>
      </span>
      {[1, 2, 3].map(p => (
        <span key={p} className={`nxo-pagination-item ${currentPage === p ? 'active' : ''}`}>
          <a className="nxo-pagination-link" onClick={() => setCurrentPage(p)}>{p}</a>
        </span>
      ))}
      <span className="nxo-pagination-item"><span className="nxo-pagination-link">...</span></span>
      <span className="nxo-pagination-item"><a className="nxo-pagination-link">10</a></span>
      <span className="nxo-pagination-item">
        <a className="nxo-pagination-link" onClick={() => setCurrentPage(p => Math.min(10, p + 1))}>
          <i className="fa-solid fa-chevron-right"></i>
        </a>
      </span>
    </nav>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatPaginatorModule, PageEvent } from '@angular/material/paginator';

@Component({
  selector: 'app-pagination',
  template: `
    <!-- Option 1: MatPaginator -->
    <mat-paginator
      class="nxo-pagination"
      [length]="100"
      [pageSize]="10"
      [pageSizeOptions]="[5, 10, 25]"
      (page)="onPageChange($event)"
      showFirstLastButtons>
    </mat-paginator>

    <!-- Option 2: Classes NXO personnalisees -->
    <nav class="nxo-pagination">
      <span [class.disabled]="currentPage === 1" class="nxo-pagination-item">
        <a class="nxo-pagination-link" (click)="goToPage(currentPage - 1)">
          <i class="fa-solid fa-chevron-left"></i>
        </a>
      </span>
      <span *ngFor="let page of visiblePages" [class.active]="currentPage === page" class="nxo-pagination-item">
        <a *ngIf="page !== '...'" class="nxo-pagination-link" (click)="goToPage(page)">{{ page }}</a>
        <span *ngIf="page === '...'" class="nxo-pagination-link">...</span>
      </span>
      <span [class.disabled]="currentPage === totalPages" class="nxo-pagination-item">
        <a class="nxo-pagination-link" (click)="goToPage(currentPage + 1)">
          <i class="fa-solid fa-chevron-right"></i>
        </a>
      </span>
    </nav>
  `
})
export class PaginationComponent {
  currentPage = 1;
  totalPages = 10;
  visiblePages = [1, 2, 3, '...', 10];

  onPageChange(event: PageEvent) {
    this.currentPage = event.pageIndex + 1;
  }

  goToPage(page: number) {
    if (page >= 1 && page <= this.totalPages) this.currentPage = page;
  }
}

// ⚠️ Exemple avec Material UI + classes NXO
import { Pagination, PaginationItem } from '@mui/material';
import { ArrowBack, ArrowForward } from '@mui/icons-material';
import { useState } from 'react';

function PaginationWithText() {
  const [page, setPage] = useState(2);

  return (
    <Pagination
      count={3}
      page={page}
      onChange={(e, value) => setPage(value)}
      className="nxo-pagination"
      renderItem={(item) => (
        <PaginationItem
          slots={{ previous: ArrowBack, next: ArrowForward }}
          {...item}
          className={`nxo-pagination-item ${item.selected ? 'active' : ''}`}
        />
      )}
    />
  );
}

export default PaginationWithText;

// ⚠️ Exemple avec Material UI + classes NXO
import { Button, Box, Typography } from '@mui/material';
import { ChevronLeft, ChevronRight } from '@mui/icons-material';
import { useState } from 'react';

function SimplePagination() {
  const [currentPage, setCurrentPage] = useState(3);
  const totalPages = 10;

  return (
    <Box sx={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
      <Button
        variant="outlined"
        className="nxo-btn nxo-btn-secondary"
        onClick={() => setCurrentPage(p => Math.max(1, p - 1))}
        disabled={currentPage === 1}
        startIcon={<ChevronLeft />}
      >
        Precedent
      </Button>
      <Typography className="nxo-text-muted">
        Page {currentPage} sur {totalPages}
      </Typography>
      <Button
        variant="outlined"
        className="nxo-btn nxo-btn-secondary"
        onClick={() => setCurrentPage(p => Math.min(totalPages, p + 1))}
        disabled={currentPage === totalPages}
        endIcon={<ChevronRight />}
      >
        Suivant
      </Button>
    </Box>
  );
}

export default SimplePagination;
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';

@Component({
  selector: 'app-simple-pagination',
  template: `
    <div style="display: flex; justify-content: space-between; align-items: center;">
      <button mat-stroked-button class="nxo-btn nxo-btn-secondary" (click)="previous()" [disabled]="currentPage === 1">
        <mat-icon>chevron_left</mat-icon> Precedent
      </button>
      <span class="nxo-text-muted">Page {{ currentPage }} sur {{ totalPages }}</span>
      <button mat-stroked-button class="nxo-btn nxo-btn-secondary" (click)="next()" [disabled]="currentPage === totalPages">
        Suivant <mat-icon>chevron_right</mat-icon>
      </button>
    </div>
  `
})
export class SimplePaginationComponent {
  currentPage = 3;
  totalPages = 10;

  previous() { if (this.currentPage > 1) this.currentPage--; }
  next() { if (this.currentPage < this.totalPages) this.currentPage++; }
}

Comportement responsive

La pagination se compacte sur mobile avec des elements plus petits. Sur tres petit ecran, les ellipses sont masquees.
Breakpoint Comportement
≤ 768px Les liens passent a 32px de hauteur avec une police reduite. flex-wrap: wrap active
≤ 480px Les .nxo-pagination-ellipsis sont masques (display: none)
