import React from 'react';
import { LinearProgress } from '@mui/material';

const ProgressExample = () => {
  return (
    <>

export default ProgressExample;

import React from 'react';

const ProgressWithLabel = () => {
  const [progress1, setProgress1] = React.useState(65);
  const [progress2, setProgress2] = React.useState({ current: 2.4, total: 4 });

  return (
    <>
      <div>
        <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '4px' }}>
          <span className="nxo-text-sm">Progression</span>
          <span className="nxo-text-sm nxo-text-muted">{progress1}%</span>
        </div>
        <div className="nxo-progress">
          <div className="nxo-progress-bar" style={{ width: `${progress1}%` }}></div>
        </div>
      </div>

      <div>
        <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '4px' }}>
          <span className="nxo-text-sm">Telechargement</span>
          <span className="nxo-text-sm nxo-text-muted">
            {progress2.current} GB / {progress2.total} GB
          </span>
        </div>
        <div className="nxo-progress">
          <div className="nxo-progress-bar nxo-progress-bar-success"
               style={{ width: `${(progress2.current / progress2.total) * 100}%` }}></div>
        </div>
      </div>
    </>
  );
};

export default ProgressWithLabel;
import { Component } from '@angular/core';

@Component({
  selector: 'app-progress-with-label',
  template: `
    <div>
      <div style="display: flex; justify-content: space-between; margin-bottom: 4px;">
        <span class="nxo-text-sm">Progression</span>
        <span class="nxo-text-sm nxo-text-muted">{{ progress1 }}%</span>
      </div>
      <div class="nxo-progress">
        <div class="nxo-progress-bar" [style.width.%]="progress1"></div>
      </div>
    </div>

    <div>
      <div style="display: flex; justify-content: space-between; margin-bottom: 4px;">
        <span class="nxo-text-sm">Telechargement</span>
        <span class="nxo-text-sm nxo-text-muted">
          {{ downloadCurrent }} GB / {{ downloadTotal }} GB
        </span>
      </div>
      <div class="nxo-progress">
        <div class="nxo-progress-bar nxo-progress-bar-success"
             [style.width.%]="downloadPercentage"></div>
      </div>
    </div>
  `
})
export class ProgressWithLabelComponent {
  progress1 = 65;
  downloadCurrent = 2.4;
  downloadTotal = 4;

  get downloadPercentage(): number {
    return (this.downloadCurrent / this.downloadTotal) * 100;
  }
}

  </div>

Comportement responsive

Les barres de progression sont naturellement flexibles. Elles occupent 100% de la largeur du conteneur parent. Aucune adaptation specifique n'est necessaire.

Voir la page Responsive pour tous les details.
