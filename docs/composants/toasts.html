import React, { useState } from 'react';
import { Snackbar, Alert } from '@mui/material';

function MyComponent() {
  const [toasts, setToasts] = useState([
    { id: 1, severity: 'success', message: 'Votre fichier a ete enregistre.' },
    { id: 2, severity: 'error', message: 'Une erreur est survenue.' },
    { id: 3, severity: 'warning', message: 'Cette action est irreversible.' },
    { id: 4, severity: 'info', message: 'Mise a jour disponible.' }
  ]);

  const handleClose = (id) => {
    setToasts(toasts.filter(t => t.id !== id));
  };

  return (
    <>
      {toasts.map(toast => (
        <Snackbar
          key={toast.id}
          open={true}
          className="nxo-toast"
          anchorOrigin={{ vertical: 'top', horizontal: 'right' }}
        >
          <Alert
            severity={toast.severity}
            className={`nxo-toast-${toast.severity}`}
            onClose={() => handleClose(toast.id)}
          >
            {toast.message}
          </Alert>
        </Snackbar>
      ))}
    </>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatSnackBar, MatSnackBarModule } from '@angular/material/snack-bar';

@Component({
  selector: 'app-my-component',
  standalone: true,
  imports: [MatSnackBarModule],
  template: `
    <button (click)="showToasts()">Afficher toasts</button>
  `
})
export class MyComponent {
  constructor(private snackBar: MatSnackBar) {}

  showToasts() {
    this.snackBar.open('Votre fichier a ete enregistre.', 'Fermer', {
      duration: 3000,
      panelClass: ['nxo-toast', 'nxo-toast-success'],
      horizontalPosition: 'right',
      verticalPosition: 'top'
    });

    this.snackBar.open('Une erreur est survenue.', 'Fermer', {
      duration: 3000,
      panelClass: ['nxo-toast', 'nxo-toast-error']
    });

    this.snackBar.open('Cette action est irreversible.', 'Fermer', {
      duration: 3000,
      panelClass: ['nxo-toast', 'nxo-toast-warning']
    });

    this.snackBar.open('Mise a jour disponible.', 'Fermer', {
      duration: 3000,
      panelClass: ['nxo-toast', 'nxo-toast-info']
    });
  }
}

// ⚠️ Exemple avec Material UI + classes NXO
import React, { useState } from 'react';
import { Snackbar, Button } from '@mui/material';

function MyComponent() {
  const [open, setOpen] = useState(false);

  const handleDelete = () => {
    setOpen(true);
  };

  const handleUndo = () => {
    console.log('Undo action');
    setOpen(false);
  };

  const handleClose = () => {
    setOpen(false);
  };

  const action = (
    <Button
      className="nxo-btn nxo-btn-sm nxo-btn-ghost"
      onClick={handleUndo}
    >
      Annuler
    </Button>
  );

  return (
    <Snackbar
      open={open}
      autoHideDuration={6000}
      onClose={handleClose}
      message="Element supprime"
      action={action}
      className="nxo-toast"
      anchorOrigin={{ vertical: 'top', horizontal: 'right' }}
    />
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatSnackBar, MatSnackBarModule } from '@angular/material/snack-bar';
import { MatButtonModule } from '@angular/material/button';

@Component({
  selector: 'app-my-component',
  standalone: true,
  imports: [MatSnackBarModule, MatButtonModule],
  template: `
    <button mat-button (click)="handleDelete()">Supprimer</button>
  `
})
export class MyComponent {
  constructor(private snackBar: MatSnackBar) {}

  handleDelete() {
    const snackBarRef = this.snackBar.open('Element supprime', 'Annuler', {
      duration: 6000,
      panelClass: 'nxo-toast',
      horizontalPosition: 'right',
      verticalPosition: 'top'
    });

    snackBarRef.onAction().subscribe(() => {
      this.handleUndo();
    });
  }

  handleUndo() {
    console.log('Undo action');
  }
}

import React, { useState } from 'react';
import { Snackbar, Button } from '@mui/material';

function MyComponent() {
  const [open, setOpen] = useState(false);

  const handleClick = () => {
    setOpen(true);
  };

  const handleClose = () => {
    setOpen(false);
  };

  // Positions disponibles: top-right, top-left, top-center,
  // bottom-right, bottom-left, bottom-center
  return (
    <>
      <Button onClick={handleClick}>Afficher toast</Button>
      <Snackbar
        open={open}
        autoHideDuration={3000}
        onClose={handleClose}
        message="Notification"
        className="nxo-toast"
        anchorOrigin={{ vertical: 'top', horizontal: 'right' }}
      />
    </>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatSnackBar, MatSnackBarModule } from '@angular/material/snack-bar';
import { MatButtonModule } from '@angular/material/button';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [MatSnackBarModule, MatButtonModule],
  template: `
    <button mat-button (click)="showToast()">Afficher toast</button>
  `
})
export class AppComponent {
  constructor(private snackBar: MatSnackBar) {}

  showToast() {
    // Positions disponibles: top, bottom (vertical)
    // start, center, end, left, right (horizontal)
    this.snackBar.open('Notification', 'Fermer', {
      duration: 3000,
      panelClass: 'nxo-toast',
      horizontalPosition: 'right',
      verticalPosition: 'top'
    });
  }
}
