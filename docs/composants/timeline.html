// ⚠️ Exemple avec Material UI + classes NXO
import { Timeline, TimelineItem, TimelineSeparator, TimelineDot, TimelineConnector, TimelineContent } from '@mui/lab';

const TimelineBasic = () => {
  const events = [
    { title: 'Commande creee', time: '10 Dec 2025, 09:00', completed: true },
    { title: 'Paiement recu', time: '10 Dec 2025, 09:15', completed: true },
    { title: 'Expedition', time: 'En attente', completed: false }
  ];

  return (
    <Timeline className="nxo-timeline">
      {events.map((event, index) => (
        <TimelineItem key={index} className="nxo-timeline-item">
          <TimelineSeparator>
            <TimelineDot className={`nxo-timeline-marker ${!event.completed ? 'nxo-timeline-marker-empty' : ''}`} />
            {index < events.length - 1 && <TimelineConnector />}
          </TimelineSeparator>
          <TimelineContent className="nxo-timeline-content">
            <div className="nxo-timeline-title">{event.title}</div>
            <div className="nxo-timeline-time">{event.time}</div>
          </TimelineContent>
        </TimelineItem>
      ))}
    </Timeline>
  );
};

// ⚠️ Exemple avec Material UI + classes NXO
import { Timeline, TimelineItem, TimelineSeparator, TimelineDot, TimelineConnector, TimelineContent } from '@mui/lab';
import { Check, Loop } from '@mui/icons-material';

const TimelineWithIcons = () => {
  const events = [
    { title: 'Compte cree', description: 'Votre compte a ete cree', time: 'Il y a 2 jours', icon: <Check />, color: 'success' },
    { title: 'Profil en cours', description: 'Completez votre profil', time: 'Maintenant', icon: <Loop className="fa-spin" />, color: 'primary' }
  ];

  return (
    <Timeline className="nxo-timeline">
      {events.map((event, index) => (
        <TimelineItem key={index} className="nxo-timeline-item has-icon">
          <TimelineSeparator>
            <TimelineDot className={`nxo-timeline-marker nxo-timeline-marker-icon nxo-timeline-marker-${event.color}`}>
              {event.icon}
            </TimelineDot>
            {index < events.length - 1 && <TimelineConnector />}
          </TimelineSeparator>
          <TimelineContent className="nxo-timeline-content">
            <div className="nxo-timeline-title">{event.title}</div>
            <div className="nxo-timeline-description">{event.description}</div>
            <div className="nxo-timeline-time">{event.time}</div>
          </TimelineContent>
        </TimelineItem>
      ))}
    </Timeline>
  );
};
