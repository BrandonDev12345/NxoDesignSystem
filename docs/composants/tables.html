const BasicTable = () => {
  const users = [
    { name: 'Jean Dupont', email: 'jean.dupont@nxo.fr', role: 'Admin', status: 'Actif', statusType: 'success' },
    { name: 'Marie Martin', email: 'marie.martin@nxo.fr', role: 'Utilisateur', status: 'Actif', statusType: 'success' },
    { name: 'Pierre Durand', email: 'pierre.durand@nxo.fr', role: 'Utilisateur', status: 'Inactif', statusType: 'neutral' }
  ];

  return (
    <table className="nxo-table">
      <thead>
        <tr>
          <th>Nom</th>
          <th>Email</th>
          <th>Role</th>
          <th>Statut</th>
        </tr>
      </thead>
      <tbody>
        {users.map((user, index) => (
          <tr key={index}>
            <td>{user.name}</td>
            <td>{user.email}</td>
            <td>{user.role}</td>
            <td>
              <span className={`nxo-badge nxo-badge-${user.statusType}`}>
                {user.status}
              </span>
            </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};

import { IconButton } from '@mui/material';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import { faEye, faPencil, faTrash } from '@fortawesome/free-solid-svg-icons';

const ActionTable = () => {
  const tickets = [
    { id: '#1234', subject: 'Probleme de connexion VPN', priority: 'Urgent', priorityType: 'error' },
    { id: '#1235', subject: 'Demande de nouveau materiel', priority: 'Moyen', priorityType: 'warning' }
  ];

  const handleView = (id) => console.log('View', id);
  const handleEdit = (id) => console.log('Edit', id);
  const handleDelete = (id) => console.log('Delete', id);

  return (
    <table className="nxo-table">
      <thead>
        <tr>
          <th>Ticket</th>
          <th>Sujet</th>
          <th>Priorite</th>
          <th>Actions</th>
        </tr>
      </thead>
      <tbody>
        {tickets.map((ticket) => (
          <tr key={ticket.id}>
            <td>{ticket.id}</td>
            <td>{ticket.subject}</td>
            <td>
              <span className={`nxo-badge nxo-badge-${ticket.priorityType}`}>
                {ticket.priority}
              </span>
            </td>
            <td>
              <div style={{ display: 'flex', gap: '4px' }}>
                <button
                  className="nxo-btn nxo-btn-icon nxo-btn-ghost nxo-btn-sm"
                  onClick={() => handleView(ticket.id)}
                >
                  <i className="fa-solid fa-eye"></i>
                </button>
                <button
                  className="nxo-btn nxo-btn-icon nxo-btn-ghost nxo-btn-sm"
                  onClick={() => handleEdit(ticket.id)}
                >
                  <i className="fa-solid fa-pencil"></i>
                </button>
                <button
                  className="nxo-btn nxo-btn-icon nxo-btn-ghost nxo-btn-sm"
                  onClick={() => handleDelete(ticket.id)}
                >
                  <i className="fa-solid fa-trash"></i>
                </button>
              </div>
            </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};

Comportement responsive

Les tableaux utilisent un wrapper .nxo-table-responsive pour activer le scroll horizontal sur mobile. Les colonnes conservent leur largeur naturelle.
Breakpoint Comportement
≤ 768px Le wrapper .nxo-table-responsive active overflow-x: auto — le tableau scrolle horizontalement. Largeur minimale du tableau : 600px
