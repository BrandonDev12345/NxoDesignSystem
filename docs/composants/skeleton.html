// ⚠️ Exemple avec Material UI + classes NXO
import { Skeleton } from '@mui/material';

const SkeletonAvatar = ({ size = 40 }) => {
  return (
    <Skeleton
      variant="circular"
      width={size}
      height={size}
      className="nxo-skeleton nxo-skeleton-circle"
    />
  );
};

// Usage:
<>
  <SkeletonAvatar size={32} />
  <SkeletonAvatar size={40} />
  <SkeletonAvatar size={48} />
  <SkeletonAvatar size={64} />
</>
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component, Input } from '@angular/core';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';

@Component({
  selector: 'app-skeleton-avatar',
  standalone: true,
  imports: [MatProgressSpinnerModule],
  template: `
    <div class="nxo-skeleton nxo-skeleton-circle"
         [ngStyle]="{width: size + 'px', height: size + 'px'}"></div>
  `
})
export class SkeletonAvatarComponent {
  @Input() size: number = 40;
}

// Usage:
<app-skeleton-avatar [size]="32"></app-skeleton-avatar>
<app-skeleton-avatar [size]="40"></app-skeleton-avatar>
<app-skeleton-avatar [size]="48"></app-skeleton-avatar>
<app-skeleton-avatar [size]="64"></app-skeleton-avatar>
<div class="nxo-skeleton nxo-skeleton-circle" style="width: {{ size }}px; height: {{ size }}px;"></div>

{# Usage in Twig: #}
{% set sizes = [32, 40, 48, 64] %}
{% for size in sizes %}
  <div class="nxo-skeleton nxo-skeleton-circle" style="width: {{ size }}px; height: {{ size }}px;"></div>
{% endfor %}

// ⚠️ Exemple avec Material UI + classes NXO
import { Skeleton } from '@mui/material';

const SkeletonListItem = () => {
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
      <Skeleton
        variant="circular"
        width={40}
        height={40}
        className="nxo-skeleton nxo-skeleton-circle"
        style={{ flexShrink: 0 }}
      />
      <div style={{ flex: 1, display: 'flex', flexDirection: 'column', gap: '8px' }}>
        <Skeleton variant="text" width="60%" height={16} className="nxo-skeleton" />
        <Skeleton variant="text" width="40%" height={14} className="nxo-skeleton" />
      </div>
    </div>
  );
};

const SkeletonList = ({ count = 3 }) => {
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
      {Array.from({ length: count }).map((_, i) => (
        <SkeletonListItem key={i} />
      ))}
    </div>
  );
};
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component, Input } from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';

@Component({
  selector: 'app-skeleton-list-item',
  standalone: true,
  imports: [MatProgressSpinnerModule],
  template: `
    <div style="display: flex; align-items: center; gap: 12px;">
      <div class="nxo-skeleton nxo-skeleton-circle"
           style="width: 40px; height: 40px; flex-shrink: 0;"></div>
      <div style="flex: 1; display: flex; flex-direction: column; gap: 8px;">
        <div class="nxo-skeleton" style="height: 16px; width: 60%;"></div>
        <div class="nxo-skeleton" style="height: 14px; width: 40%;"></div>
      </div>
    </div>
  `
})
export class SkeletonListItemComponent {}

@Component({
  selector: 'app-skeleton-list',
  standalone: true,
  imports: [CommonModule, SkeletonListItemComponent],
  template: `
    <div style="display: flex; flex-direction: column; gap: 16px;">
      <app-skeleton-list-item *ngFor="let item of items"></app-skeleton-list-item>
    </div>
  `
})
export class SkeletonListComponent {
  @Input() count: number = 3;
  get items() { return Array(this.count).fill(0); }
}

Comportement responsive

Les Skeletons sont naturellement flexibles. Ils occupent 100% de la largeur disponible et s'adaptent au layout parent. Aucune adaptation specifique n'est necessaire.

Voir la page Responsive pour tous les details.
