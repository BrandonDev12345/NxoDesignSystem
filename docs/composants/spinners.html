// ⚠️ Exemple avec Material UI + classes NXO
import { useState } from 'react';
import { CircularProgress } from '@mui/material';

function LoadingButton() {
  const [loading, setLoading] = useState(true);

  return (
    <>
      <button className="nxo-btn nxo-btn-primary" disabled={loading}>
        <CircularProgress className="nxo-spinner nxo-spinner-sm nxo-spinner-white" size={16} />
        Chargement...
      </button>

      <button className="nxo-btn nxo-btn-secondary" disabled={loading}>
        <CircularProgress className="nxo-spinner nxo-spinner-sm" size={16} />
        Patientez...
      </button>
    </>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';

@Component({
  selector: 'app-loading-button',
  standalone: true,
  imports: [MatProgressSpinnerModule],
  template: `
    <button class="nxo-btn nxo-btn-primary" [disabled]="loading">
      <mat-spinner class="nxo-spinner nxo-spinner-sm nxo-spinner-white" [diameter]="16"></mat-spinner>
      Chargement...
    </button>

    <button class="nxo-btn nxo-btn-secondary" [disabled]="loading">
      <mat-spinner class="nxo-spinner nxo-spinner-sm" [diameter]="16"></mat-spinner>
      Patientez...
    </button>
  `
})
export class LoadingButtonComponent {
  loading = true;
}

<button class="nxo-btn nxo-btn-primary" {{ loading ? 'disabled' : '' }}>

  <span class="nxo-spinner nxo-spinner-sm"></span>

Overlay de chargement

Preview

Contenu de la page...

<div style="position: relative;">
  <!-- Your content -->
  <div class="nxo-loading-overlay">
    <div class="nxo-spinner nxo-spinner-lg nxo-spinner-primary"></div>
  </div>
</div>
// ⚠️ Exemple avec Material UI + classes NXO
import { useState } from 'react';
import { CircularProgress } from '@mui/material';

function LoadingOverlay() {
  const [loading, setLoading] = useState(true);

  return (
    <div style={{ position: 'relative' }}>
      <div style={{ padding: '16px' }}>
        <p>Contenu de la page...</p>
      </div>
      {loading && (
        <div className="nxo-loading-overlay">
          <CircularProgress className="nxo-spinner nxo-spinner-lg nxo-spinner-primary" size={40} />
        </div>
      )}
    </div>
  );
}
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';

@Component({
  selector: 'app-loading-overlay',
  standalone: true,
  imports: [CommonModule, MatProgressSpinnerModule],
  template: `
    <div style="position: relative;">
      <div style="padding: 16px;">
        <p>Contenu de la page...</p>
      </div>
      <div class="nxo-loading-overlay" *ngIf="loading">
        <mat-spinner class="nxo-spinner nxo-spinner-lg nxo-spinner-primary" [diameter]="40"></mat-spinner>
      </div>
    </div>
  `
})
export class LoadingOverlayComponent {
  loading = true;
}
{# Twig template #}
<div style="position: relative;">
  <div style="padding: 16px;">
    <p>Contenu de la page...</p>
  </div>
  {% if loading %}
  <div class="nxo-loading-overlay">
    <div class="nxo-spinner nxo-spinner-lg nxo-spinner-primary"></div>
  </div>
  {% endif %}
</div>

Comportement responsive

Les Spinners sont naturellement flexibles. Leurs dimensions fixes s'integrent dans tous les contextes responsive. Aucune adaptation specifique n'est necessaire.

Voir la page Responsive pour tous les details.
