// ⚠️ Exemple avec Material UI + classes NXO
import { List, ListItem, ListItemAvatar, ListItemText, Avatar } from '@mui/material';

const users = [
  { initials: 'JD', name: 'Jean Dupont', email: 'jean.dupont@nxo.fr', color: 'var(--fayat-blue)' },
  { initials: 'ML', name: 'Marie Laurent', email: 'marie.laurent@nxo.fr', color: 'var(--nxo-color-success)' },
  { initials: 'PK', name: 'Pierre Klein', email: 'pierre.klein@nxo.fr', color: 'var(--nxo-color-warning)' }
];

const AvatarList = () => (
  <List className="nxo-list">
    {users.map(user => (
      <ListItem key={user.email} className="nxo-list-item">
        <ListItemAvatar>
          <Avatar
            className="nxo-avatar nxo-avatar-sm"
            sx={{ bgcolor: user.color }}
          >
            {user.initials}
          </Avatar>
        </ListItemAvatar>
        <ListItemText
          primary={user.name}
          secondary={user.email}
          primaryTypographyProps={{ style: { fontWeight: 500 } }}
          secondaryTypographyProps={{ className: 'nxo-text-sm nxo-text-muted' }}
        />
      </ListItem>
    ))}
  </List>
);
// ⚠️ Exemple avec Angular Material + classes NXO

  <li class="nxo-list-item nxo-list-item-interactive">
    <span>Securite</span>
    <i class="fa-solid fa-chevron-right"></i>
  </li>
</ul>
// ⚠️ Exemple avec Material UI + classes NXO
import { List, ListItem, ListItemButton, ListItemText } from '@mui/material';
import { ChevronRight } from '@mui/icons-material';

const menuItems = [
  'Parametres du compte',
  'Notifications',
  'Securite'
];

const InteractiveList = () => (
  <List className="nxo-list nxo-list-bordered">
    {menuItems.map(item => (
      <ListItem key={item} disablePadding className="nxo-list-item nxo-list-item-interactive">
        <ListItemButton onClick={() => console.log(item)}>
          <ListItemText primary={item} />
          <ChevronRight sx={{ color: 'var(--nxo-text-muted)' }} />
        </ListItemButton>
      </ListItem>
    ))}
  </List>
);
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatListModule } from '@angular/material/list';
import { MatIconModule } from '@angular/material/icon';

@Component({
  selector: 'app-interactive-list',
  standalone: true,
  imports: [CommonModule, MatListModule, MatIconModule],
  template: `
    <mat-nav-list class="nxo-list nxo-list-bordered">
      <mat-list-item
        *ngFor="let item of menuItems"
        class="nxo-list-item nxo-list-item-interactive"
        (click)="handleItemClick(item)"
      >
        <span matListItemTitle>{{ item }}</span>
        <mat-icon matListItemMeta style="color: var(--nxo-text-muted);">chevron_right</mat-icon>
      </mat-list-item>
    </mat-nav-list>
  `
})
export class InteractiveListComponent {
  menuItems = [
    'Parametres du compte',
    'Notifications',
    'Securite'
  ];

  handleItemClick(item: string) {
    console.log(item);
  }
}
