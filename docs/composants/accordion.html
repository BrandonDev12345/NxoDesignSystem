Accordion

Composant pliable pour organiser le contenu en sections extensibles.

Accordion basique

Preview

NXO Design System est une bibliotheque de composants UI pour creer des interfaces coherentes et accessibles.

Incluez simplement les fichiers CSS tokens.css et components.css dans votre projet.

Oui, NXO est compatible avec Angular, React, Vue et tout autre framework JavaScript.

// ⚠️ Exemple avec Material UI + classes NXO
import { Accordion, AccordionSummary, AccordionDetails, Typography } from '@mui/material';
import ExpandMoreIcon from '@mui/icons-material/ExpandMore';

const items = [
  {
    title: "Qu'est-ce que NXO Design System ?",
    content: "NXO Design System est une bibliotheque de composants..."
  },
  {
    title: "Comment l'installer ?",
    content: "Incluez simplement les fichiers CSS..."
  },
  {
    title: "Est-il compatible avec les frameworks ?",
    content: "Oui, NXO est compatible avec Angular, React, Vue..."
  }
];

const AccordionBasic = () => {
  return (
    <div className="nxo-accordion">
      {items.map((item, index) => (
        <Accordion key={index} defaultExpanded={index === 0} className="nxo-accordion-item">
          <AccordionSummary
            expandIcon={<ExpandMoreIcon />}
            className="nxo-accordion-header"
          >
            <Typography>{item.title}</Typography>
          </AccordionSummary>
          <AccordionDetails className="nxo-accordion-content">
            <Typography className="nxo-text">{item.content}</Typography>
          </AccordionDetails>
        </Accordion>
      ))}
    </div>
  );
};

export default AccordionBasic;
// ⚠️ Exemple avec Angular Material + classes NXO
import { Component } from '@angular/core';
import { MatExpansionModule } from '@angular/material/expansion';

@Component({
  selector: 'app-accordion',
  standalone: true,
  imports: [MatExpansionModule],
  template: `
    <mat-accordion class="nxo-accordion">
      @for (item of items; track item.title; let i = $index) {
        <mat-expansion-panel [expanded]="i === 0" class="nxo-accordion-item">
          <mat-expansion-panel-header class="nxo-accordion-header">
            <mat-panel-title>{{ item.title }}</mat-panel-title>
          </mat-expansion-panel-header>
          <p class="nxo-text">{{ item.content }}</p>
        </mat-expansion-panel>
      }
    </mat-accordion>
  `
})

      </div>
    </div>
  <?php endforeach; ?>
</div>

// ⚠️ Exemple avec Material UI + classes NXO
import { Accordion, AccordionSummary, AccordionDetails, Typography, Box } from '@mui/material';
import ExpandMoreIcon from '@mui/icons-material/ExpandMore';
import PersonIcon from '@mui/icons-material/Person';
import SecurityIcon from '@mui/icons-material/Security';
import NotificationsIcon from '@mui/icons-material/Notifications';

const items = [
  {
    icon: <PersonIcon sx={{ color: 'var(--fayat-blue)', mr: 1 }} />,
    title: "Informations personnelles",
    content: "Gerez vos informations de profil et vos preferences."
  },
  {
    icon: <SecurityIcon sx={{ color: 'var(--nxo-color-success)', mr: 1 }} />,
    title: "Securite",
    content: "Configurez vos parametres de securite et d'authentification."
  },
  {
    icon: <NotificationsIcon sx={{ color: 'var(--nxo-color-warning)', mr: 1 }} />,
    title: "Notifications",
    content: "Gerez vos preferences de notifications."
  }
];

const AccordionWithIcons = () => {
  return (
    <div className="nxo-accordion">
      {items.map((item, index) => (
        <Accordion key={index} defaultExpanded={index === 0} className="nxo-accordion-item">
          <AccordionSummary
            expandIcon={<ExpandMoreIcon />}
            className="nxo-accordion-header"
          >
            <Box sx={{ display: 'flex', alignItems: 'center' }}>
              {item.icon}
              <Typography>{item.title}</Typography>
            </Box>
          </AccordionSummary>
          <AccordionDetails className="nxo-accordion-content">
            <Typography className="nxo-text">{item.content}</Typography>
          </AccordionDetails>
        </Accordion>
      ))}
    </div>
  );
};

export default AccordionWithIcons;
