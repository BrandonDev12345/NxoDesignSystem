Header Responsive Officiel

Header unifie pour l'Espace Client NXO avec 5 breakpoints responsive. C'est le header a utiliser pour tous les portails clients.

Structure en 3 zones

<!-- Header NXO Responsive -->
<header class="nxo-header">
  <!-- Zone gauche: Menu + Logo + Nom service -->
  <div class="nxo-header__zone-left">
    <button class="nxo-header__icon nxo-header__icon--apps">...</button>
    <button class="nxo-header__icon nxo-header__icon--burger">...</button>
    <div class="nxo-header__logo"><img src="logo.png"></div>
    <span class="nxo-header__service-name">Nom Service</span>
  </div>

  <!-- Zone centre: Search bar (ou logo sur mobile) -->
  <div class="nxo-header__zone-center">
    <div class="nxo-header__search">...</div>
    <div class="nxo-header__logo nxo-header__logo--mobile">...</div>
  </div>

  <!-- Zone droite: Lang + Actions + Profil -->
  <div class="nxo-header__zone-right">
    <div class="nxo-header__lang">...</div>
    <button class="nxo-header__icon nxo-header__icon--search">...</button>
    <button class="nxo-header__icon nxo-header__icon--expand">...</button>
    <button class="nxo-header__icon nxo-header__icon--help">...</button>
    <button class="nxo-header__icon nxo-header__icon--profile">...</button>
  </div>
</header>
